//  Write a function to apply the Bubble Sort algorithm to a given array.
 
function bubbleSort(arr) {
    let swap;
    let n = arr.length - 1;
    let x = arr;
    do {
        swap = false;
        for (let i = 0; i < n; i++) {
            if (x[i] < x[i + 1]) {
                let value = x[i];
                x[i] = x[i + 1];
                x[i + 1] = value;
                swap = true;
            }
        }
        n--;
    } while (swap);
    return x;
}
